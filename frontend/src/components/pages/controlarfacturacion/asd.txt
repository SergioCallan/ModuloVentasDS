import React from 'react';

// Componente para facturas individuales
const FacturaIndividual = ({ factura }) => {
  return (
    <div className="factura-individual">
      <p>ID de Factura: {factura.id}</p>
      <p>Monto: {factura.monto}</p>
      {/* M치s detalles de la factura */}
    </div>
  );
};

// Componente que representa un grupo de facturas (puede incluir subgrupos)
const GrupoFacturas = ({ facturas }) => {
  return (
    <div className="grupo-facturas">
      {facturas.map((factura, index) =>
        Array.isArray(factura) ? (
          <GrupoFacturas key={index} facturas={factura} />
        ) : (
          <FacturaIndividual key={factura.id} factura={factura} />
        )
      )}
    </div>
  );
};

// Componente principal que utiliza GrupoFacturas
const Facturacion = () => {
  // Ejemplo de datos que representan una estructura de facturaci칩n jer치rquica
  const datosDeFacturacion = [
    { id: 'F001', monto: 100.0 },
    [
      // Subgrupo de facturas
      { id: 'F002', monto: 150.0 },
      { id: 'F003', monto: 200.0 },
    ],
    { id: 'F004', monto: 250.0 },
  ];

  return (
    <div className="facturacion">
      <h1>Facturaci칩n</h1>
      <GrupoFacturas facturas={datosDeFacturacion} />
    </div>
  );
};

export default Facturacion;
